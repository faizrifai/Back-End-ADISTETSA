class CreateDataRiwayatGajiBerkalaTestCase(SetupData):

        self.assertEqual(response.status_code, status.HTTP_201_CREATED)

    def test_edit_data_riwayat_gaji_berkala(self):
        # add data riwayat_gaji_berkala
        self.client.post(reverse('data_guru_riwayat_gaji_berkala'), self.data_riwayat_gaji_berkala)

        # test edit data
        patch_response = self.client.patch(reverse('data_guru_riwayat_gaji_berkala', kwargs={'pk': 1}), self.data_riwayat_gaji_berkala)
        put_response = self.client.put(reverse('data_guru_riwayat_gaji_berkala', kwargs={'pk': 1}), self.data_riwayat_gaji_berkala)

        self.assertEqual(patch_response.status_code, status.HTTP_200_OK)
        self.assertEqual(put_response.status_code, status.HTTP_200_OK)

    def test_delete_data_riwayat_gaji_berkala(self):
        # add data riwayat_gaji_berkala
        self.client.post(reverse('data_guru_riwayat_gaji_berkala'), self.data_riwayat_gaji_berkala)

        # test delete data
        response = self.client.delete(reverse('data_guru_riwayat_gaji_berkala', kwargs={'pk': 1}))
